用webpack配置react项目开发

新建webpack.config.js配置文件

module.exports = {
  entry: './index.js',//入口文件
  output: {
    filename: 'bundle.js'//目标输出文件
  },
  module: {
    loaders:[
      {
        test: /\.js[x]?$/,//
        exclude: /node_modules/,
        loader: 'babel',
        query: {
                    presets: ['es2015','react']//转换es6到es5并转换jsx到js的配置
                }
      }
    ]
  }
};

新建package.json并配置

{
  "name": "es6",
  "version": "1.0.0",
  "description": "",
  "main": "index.js",
  "dependencies": {  //以下依赖包必须，应该能看懂
    "babel-core": "^6.13.2",
    "babel-loader": "^6.2.4",
    "babel-preset-es2015": "^6.13.2",
    "babel-preset-react": "^6.11.1",
    "react-dom": "^15.3.0",
    "react": "^15.3.0"
  },
  "devDependencies": {
    "react-dom": "^15.3.0"
  },
  "scripts": {
    "test": "echo \"Error: no test specified\" && exit 1"
  },
  "author": "",
  "license": "ISC"
}

编辑index.html

//index.html
<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <title>es6</title>
    <link rel="stylesheet" href="">
</head>
<body>
    <div class="content"></div>
    <script type="text/javascript" src="bundle.js"></script>
</body>
</html>

编辑inex.js

//index.js
import Components from "./Components.jsx"

var ReactDOM = require('react-dom');
var React = require('react');

ReactDOM.render(
        <Components />,
        document.getElementsByClassName("content")[0]
)

编辑Components.jsx

var React = require('react');

var Components = React.createClass({

    render: function() {
        return (
            <div>hello world！</div>
        );
    }

});
export default Components;

安装依赖并监听文件变化

npm install & webpack -w

下面你就可以在入口文件内添加es6代码或者jsx代码了。并且可以相互调用，最终会打包到bundle.js文件里。项目里引入bundle.js就好，试试吧。
